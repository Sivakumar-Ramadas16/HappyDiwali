package com.example.rndmsnew.Adapter

import android.support.v7.widget.RecyclerView
import android.view.LayoutInflater
import android.view.View
import android.view.ViewGroup
import android.widget.ImageView
import android.widget.TextView
import com.example.rndmsnew.Model.Thought
import com.example.rndmsnew.NUM_LIKES
import com.example.rndmsnew.R
import com.example.rndmsnew.THOUGHT_REF
import com.google.firebase.firestore.FirebaseFirestore
import kotlinx.android.synthetic.main.row_thought.view.*
import java.text.DateFormat
import java.text.SimpleDateFormat
import java.util.*
import kotlin.collections.ArrayList

class ThoughtAdapter(val thoughts: ArrayList<Thought>) : RecyclerView.Adapter<ThoughtAdapter.ViewHolder>() {
    override fun onCreateViewHolder(parent: ViewGroup, position: Int): ViewHolder {
       val view = LayoutInflater.from(parent.context).inflate(R.layout.row_thought,parent,false)
       return ViewHolder(view)
    }

    override fun getItemCount(): Int {
        return thoughts.count()
    }

    override fun onBindViewHolder(holder: ViewHolder, position: Int) {
        holder.bindThought(thoughts[position])
    }

    inner class ViewHolder(itemView:View): RecyclerView.ViewHolder(itemView){
        val  username = itemView?.findViewById<TextView>(R.id.RowUserName)
        val  timestamp = itemView?.findViewById<TextView>(R.id.RowTimeStamp)
        val  thoughtTxt = itemView?.findViewById<TextView>(R.id.RowThought)
        val  numLikes = itemView?.findViewById<TextView>(R.id.RowNumLikes)
        val  numComments = itemView?.findViewById<TextView>(R.id.RowNumComments)


        val  imageLikes = itemView?.findViewById<ImageView>(R.id.RowImageLikes)
        val  imagecomments = itemView?.findViewById<ImageView>(R.id.RowImageComment)
        val  imageEcclipse = itemView?.findViewById<ImageView>(R.id.RowImageEcllipse)

         fun bindThought(t :Thought) {

             username?.text = t.username.toString()
             thoughtTxt?.text = t.thoughtTxt.toString()
             numLikes?.text = t.numLikes.toString()
             val DateFormat = SimpleDateFormat("MMM d, h:mm a",Locale.getDefault())
             val dateString = DateFormat.format(t.timestamp)
             timestamp?.text = dateString

             imageLikes.setOnClickListener{
                 FirebaseFirestore.getInstance().collection(THOUGHT_REF).document(t.documentId)
                     .update(NUM_LIKES, t.numLikes+1)

             }
        }

    }
}
